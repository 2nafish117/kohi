1188 // Calculates a rotation matrix based on the quaternion and the passed in center point.
1189 KINLINE mat4 quat_to_rotation_matrix(quat q, vec3 center) {
1190  mat4 out_matrix;
1191 
1192  f32* o = out_matrix.data;
1193  o[0] = (q.x * q.x) - (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1194  o[1] = 2.0f * ((q.x * q.y) + (q.z * q.w));
1195  o[2] = 2.0f * ((q.x * q.z) - (q.y * q.w));
1196  o[3] = center.x - center.x * o[0] - center.y * o[1] - center.z * o[2];
1197 
1198  o[4] = 2.0f * ((q.x * q.y) - (q.z * q.w));
1199  o[5] = -(q.x * q.x) + (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1200  o[6] = 2.0f * ((q.y * q.z) + (q.x * q.w));
1201  o[7] = center.y - center.x * o[4] - center.y * o[5] - center.z * o[6];
1202 
1203  o[8] = 2.0f * ((q.x * q.z) + (q.y * q.w));
1204  o[9] = 2.0f * ((q.y * q.z) - (q.x * q.w));
1205  o[10] = -(q.x * q.x) - (q.y * q.y) + (q.z * q.z) + (q.w * q.w);
1206  o[11] = center.z - center.x * o[8] - center.y * o[9] - center.z * o[10];
1207 
1208  o[12] = 0.0f;
1209  o[13] = 0.0f;
1210  o[14] = 0.0f;
1211  o[15] = 1.0f;
1212  return out_matrix;
1213 }

1227 KINLINE quat quat_slerp(quat q_0, quat q_1, f32 percentage) {
1228  quat out_quaternion;
1229  // Source: https://en.wikipedia.org/wiki/Slerp
1230  // Only unit quaternions are valid rotations.
1231  // Normalize to avoid undefined behavior.
1232  quat v0 = quat_normalize(q_0);
1233  quat v1 = quat_normalize(q_1);
1234 
1235  // Compute the cosine of the angle between the two vectors.
1236  f32 dot = quat_dot(v0, v1);
1237 
1238  // If the dot product is negative, slerp won't take
1239  // the shorter path. Note that v1 and -v1 are equivalent when
1240  // the negation is applied to all four components. Fix by
1241  // reversing one quaternion.
1242  if (dot < 0.0f) {
1243  v1.x = -v1.x;
1244  v1.y = -v1.y;
1245  v1.z = -v1.z;
1246  v1.w = -v1.w;
1247  dot = -dot;
1248  }
1249 
1250  const f32 DOT_THRESHOLD = 0.9995f;
1251  if (dot > DOT_THRESHOLD) {
1252  // If the inputs are too close for comfort, linearly interpolate
1253  // and normalize the result.
1254  out_quaternion = (quat){
1255  v0.x + ((v1.x - v0.x) * percentage),
1256  v0.y + ((v1.y - v0.y) * percentage),
1257  v0.z + ((v1.z - v0.z) * percentage),
1258  v0.w + ((v1.w - v0.w) * percentage)};
1259 
1260  return quat_normalize(out_quaternion);
1261  }
1262 
1263  // Since dot is in range [0, DOT_THRESHOLD], acos is safe
1264  f32 theta_0 = kacos(dot); // theta_0 = angle between input vectors
1265  f32 theta = theta_0 * percentage; // theta = angle between v0 and result
1266  f32 sin_theta = ksin(theta); // compute this value only once
1267  f32 sin_theta_0 = ksin(theta_0); // compute this value only once
1268 
1269  f32 s0 = kcos(theta) - dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
1270  f32 s1 = sin_theta / sin_theta_0;
1271 
1272  return (quat){
1273  (v0.x * s0) + (v1.x * s1),
1274  (v0.y * s0) + (v1.y * s1),
1275  (v0.z * s0) + (v1.z * s1),
1276  (v0.w * s0) + (v1.w * s1)};
1277 }
