1394 KINLINE mat4 quat_to_rotation_matrix(quat q, vec3 center) {
1395  mat4 out_matrix;

1397  f32* o = out_matrix.data;
1398  o[0] = (q.x * q.x) - (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1399  o[1] = 2.0f * ((q.x * q.y) + (q.z * q.w));
1400  o[2] = 2.0f * ((q.x * q.z) - (q.y * q.w));
1401  o[3] = center.x - center.x * o[0] - center.y * o[1] - center.z * o[2];
1402 
1403  o[4] = 2.0f * ((q.x * q.y) - (q.z * q.w));
1404  o[5] = -(q.x * q.x) + (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1405  o[6] = 2.0f * ((q.y * q.z) + (q.x * q.w));
1406  o[7] = center.y - center.x * o[4] - center.y * o[5] - center.z * o[6];
1407 
1408  o[8] = 2.0f * ((q.x * q.z) + (q.y * q.w));
1409  o[9] = 2.0f * ((q.y * q.z) - (q.x * q.w));
1410  o[10] = -(q.x * q.x) - (q.y * q.y) + (q.z * q.z) + (q.w * q.w);
1411  o[11] = center.z - center.x * o[8] - center.y * o[9] - center.z * o[10];
1412 
1413  o[12] = 0.0f;
1414  o[13] = 0.0f;
1415  o[14] = 0.0f;

1449 KINLINE quat quat_slerp(quat q_0, quat q_1, f32 percentage) {
1450  quat out_quaternion;
1451  // Source: https://en.wikipedia.org/wiki/Slerp
1452  // Only unit quaternions are valid rotations.
1453  // Normalize to avoid undefined behavior.
1454  quat v0 = quat_normalize(q_0);
1455  quat v1 = quat_normalize(q_1);
1456 
1457  // Compute the cosine of the angle between the two vectors.
1458  f32 dot = quat_dot(v0, v1);
1459 
1460  // If the dot product is negative, slerp won't take
1461  // the shorter path. Note that v1 and -v1 are equivalent when
1462  // the negation is applied to all four components. Fix by
1463  // reversing one quaternion.
1464  if (dot < 0.0f) {
1465  v1.x = -v1.x;
1466  v1.y = -v1.y;
1467  v1.z = -v1.z;
1468  v1.w = -v1.w;
1469  dot = -dot;
1470  }
1471 
1472  const f32 DOT_THRESHOLD = 0.9995f;
1473  if (dot > DOT_THRESHOLD) {
1474  // If the inputs are too close for comfort, linearly interpolate
1475  // and normalize the result.
1476  out_quaternion = (quat){
1477  v0.x + ((v1.x - v0.x) * percentage),
1478  v0.y + ((v1.y - v0.y) * percentage),
1479  v0.z + ((v1.z - v0.z) * percentage),
1480  v0.w + ((v1.w - v0.w) * percentage)};
1481 
1482  return quat_normalize(out_quaternion);
1483  }
1484 
1485  // Since dot is in range [0, DOT_THRESHOLD], acos is safe
1486  f32 theta_0 = kacos(dot); // theta_0 = angle between input vectors
1487  f32 theta = theta_0 * percentage; // theta = angle between v0 and result
1488  f32 sin_theta = ksin(theta); // compute this value only once
1489  f32 sin_theta_0 = ksin(theta_0); // compute this value only once
1490 
1491  f32 s0 = kcos(theta) - dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
1492  f32 s1 = sin_theta / sin_theta_0;
1493 
1494  return (quat){
1495  (v0.x * s0) + (v1.x * s1),
1496  (v0.y * s0) + (v1.y * s1),
1497  (v0.z * s0) + (v1.z * s1),
1498  (v0.w * s0) + (v1.w * s1)};
1499 }
